## 🔍 SELECT — View Data

```sql
-- View all employees
SELECT * FROM EMPLOYEE;

-- View specific columns only
SELECT ID, NAME, EMAIL FROM EMPLOYEE;

-- View only fully onboarded employees (all 3 done)
SELECT * FROM EMPLOYEE
WHERE DOCUMENT_SUBMITTED = TRUE
  AND IT_SETUP_DONE = TRUE
  AND TRAINING_COMPLETED = TRUE;

-- View employees with pending tasks
SELECT * FROM EMPLOYEE
WHERE DOCUMENT_SUBMITTED = FALSE
   OR IT_SETUP_DONE = FALSE
   OR TRAINING_COMPLETED = FALSE;

-- Search by name
SELECT * FROM EMPLOYEE WHERE NAME = 'Alice Johnson';

-- Search by email
SELECT * FROM EMPLOYEE WHERE EMAIL = 'alice@company.com';

-- Count total employees
SELECT COUNT(*) AS TOTAL FROM EMPLOYEE;

-- Count how many completed each step
SELECT
  COUNT(*) AS TOTAL,
  SUM(CASE WHEN DOCUMENT_SUBMITTED = TRUE THEN 1 ELSE 0 END) AS DOCS_DONE,
  SUM(CASE WHEN IT_SETUP_DONE      = TRUE THEN 1 ELSE 0 END) AS IT_DONE,
  SUM(CASE WHEN TRAINING_COMPLETED = TRUE THEN 1 ELSE 0 END) AS TRAINING_DONE
FROM EMPLOYEE;

-- Order by name A-Z
SELECT * FROM EMPLOYEE ORDER BY NAME ASC;
```

## ✏️ UPDATE — Modify Records

```sql
-- Mark document submitted for employee ID 1
UPDATE EMPLOYEE SET DOCUMENT_SUBMITTED = TRUE WHERE ID = 1;

-- Mark IT setup done for employee ID 1
UPDATE EMPLOYEE SET IT_SETUP_DONE = TRUE WHERE ID = 1;

-- Mark training completed for employee ID 1
UPDATE EMPLOYEE SET TRAINING_COMPLETED = TRUE WHERE ID = 1;

-- Fully onboard an employee (all steps at once)
UPDATE EMPLOYEE
SET DOCUMENT_SUBMITTED = TRUE,
    IT_SETUP_DONE = TRUE,
    TRAINING_COMPLETED = TRUE
WHERE ID = 2;

-- Update name and email
UPDATE EMPLOYEE
SET NAME = 'Bob Johnson', EMAIL = 'bob.j@company.com'
WHERE ID = 2;

-- Reset all steps for an employee
UPDATE EMPLOYEE
SET DOCUMENT_SUBMITTED = FALSE,
    IT_SETUP_DONE = FALSE,
    TRAINING_COMPLETED = FALSE
WHERE ID = 3;
```

## 🏗️ TABLE — Structure Commands

```sql
-- View table structure
SHOW COLUMNS FROM EMPLOYEE;

-- View all tables in the database
SHOW TABLES;

-- Drop the table entirely (⚠️ DANGEROUS — deletes all data!)
DROP TABLE IF EXISTS EMPLOYEE;

-- Recreate the table from scratch
CREATE TABLE IF NOT EXISTS EMPLOYEE (
  ID                  BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  NAME                VARCHAR(255),
  EMAIL               VARCHAR(255),
  DOCUMENT_SUBMITTED  BOOLEAN NOT NULL DEFAULT FALSE,
  IT_SETUP_DONE       BOOLEAN NOT NULL DEFAULT FALSE,
  TRAINING_COMPLETED  BOOLEAN NOT NULL DEFAULT FALSE
);
```

## 📊 REPORTING — Useful Queries

```sql
-- Progress report for each employee
SELECT
  ID,
  NAME,
  EMAIL,
  CASE WHEN DOCUMENT_SUBMITTED = TRUE THEN 'Done' ELSE 'Pending' END AS DOCS,
  CASE WHEN IT_SETUP_DONE      = TRUE THEN 'Done' ELSE 'Pending' END AS IT,
  CASE WHEN TRAINING_COMPLETED = TRUE THEN 'Done' ELSE 'Pending' END AS TRAINING,
  (CASE WHEN DOCUMENT_SUBMITTED = TRUE THEN 1 ELSE 0 END +
   CASE WHEN IT_SETUP_DONE      = TRUE THEN 1 ELSE 0 END +
   CASE WHEN TRAINING_COMPLETED = TRUE THEN 1 ELSE 0 END) AS STEPS_COMPLETED
FROM EMPLOYEE
ORDER BY STEPS_COMPLETED DESC;

-- Employees sorted by completion percentage
SELECT
  NAME,
  ROUND(
    (CASE WHEN DOCUMENT_SUBMITTED = TRUE THEN 1 ELSE 0 END +
     CASE WHEN IT_SETUP_DONE      = TRUE THEN 1 ELSE 0 END +
     CASE WHEN TRAINING_COMPLETED = TRUE THEN 1 ELSE 0 END) * 100.0 / 3
  , 0) AS PROGRESS_PCT
FROM EMPLOYEE
ORDER BY PROGRESS_PCT DESC;

-- Summary dashboard stats
SELECT
  COUNT(*)                                                           AS TOTAL_EMPLOYEES,
  SUM(CASE WHEN DOCUMENT_SUBMITTED = TRUE  THEN 1 ELSE 0 END)       AS DOCS_SUBMITTED,
  SUM(CASE WHEN IT_SETUP_DONE = TRUE       THEN 1 ELSE 0 END)       AS IT_SETUP_DONE,
  SUM(CASE WHEN TRAINING_COMPLETED = TRUE  THEN 1 ELSE 0 END)       AS TRAINING_DONE,
  SUM(CASE WHEN DOCUMENT_SUBMITTED = TRUE
            AND IT_SETUP_DONE = TRUE
            AND TRAINING_COMPLETED = TRUE  THEN 1 ELSE 0 END)       AS FULLY_ONBOARDED
FROM EMPLOYEE;
```
